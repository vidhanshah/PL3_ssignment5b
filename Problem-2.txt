	<section>
		<h2> &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;
      &nbsp;&nbsp;&nbsp;&nbsp;
      Program</h2>
		<code>
			#include &lt stdio.h &gt<br>
			void main()<br>
			{<br>
	    	&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;int num,r,sum,temp;<br>
	    	&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;int stno,enno;<br>
	    
	        &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;printf("Input starting number of range: ");<br>
	        &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;scanf("%d",&stno);<br>
	        &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;printf("Input ending number of range : ");<br>
	        &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;scanf("%d",&enno);<br>
	   
	        &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;printf("Armstrong numbers in given range are: ");<br>
	        &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;for(num=stno;num<=enno;num++)<br>
	    	&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;{<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;temp=num;<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;sum=0;<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;while(temp!=0)<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;{<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;r=temp % 10;<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;temp=temp/10;<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;sum=sum+(r*r*r);<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;if(sum==num)<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;printf("%d ",num);<br>
		    &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;}<br>

	    	&nbsp;&nbsp;&nbsp;printf("\n");<br>
			}

		</code>
